Embed TREF in Your Website

Copy-paste snippets to add TREF blocks in seconds

Step 1: Add the Script

8 KB

Add this script tag to your HTML. Include it once, then use TREF anywhere on your page.

HTML (head or end of body)
<script src="https://tref.lpmwfx.com/js/tref.min.js"></script>

Step 2: Display a TREF Block

Easy

Create a block and display it with automatic styling.

JavaScript
// Create a block
const block = await TREF.publish("Your knowledge content here", {
  author: "Your Name",
  license: "CC-BY-4.0"
});

// Wrap and display it
const wrapper = new TREF.TrefWrapper(block);
document.getElementById("my-block").innerHTML = wrapper.toHTML();
HTML
<!-- Container for the block -->
<div id="my-block"></div>

<!-- Include the styles -->
<style>
  /* Paste this once in your CSS */
  .tref-wrapper { font-family: system-ui; border: 1px solid #e5e7eb; border-radius: 8px; padding: 12px; background: #fafafa; max-width: 400px; }
  .tref-header { display: flex; align-items: center; gap: 8px; margin-bottom: 8px; }
  .tref-icon { display: inline-flex; width: 24px; height: 24px; }
  .tref-icon svg { width: 100%; height: 100%; }
  .tref-id { font-family: monospace; font-size: 12px; color: #6b7280; background: #e5e7eb; padding: 2px 6px; border-radius: 4px; }
  .tref-meta { font-size: 12px; color: #9ca3af; margin-left: auto; }
  .tref-content { font-size: 14px; line-height: 1.5; color: #374151; white-space: pre-wrap; }
</style>

Step 3: Receive TREF Blocks (Drop Zone)

Easy

Let users drag TREF blocks into your page.

HTML + JavaScript
<!-- The drop zone element -->
<div id="my-dropzone" data-placeholder="Drop TREF block here">
  Drop TREF block here
</div>

<script>
  // Create receiver
  const receiver = new TREF.TrefReceiver(
    document.getElementById("my-dropzone"),
    {
      onReceive: (wrapper) => {
        console.log("Received:", wrapper.id);
        receiver.showBlock(wrapper);
      },
      onError: (err) => {
        console.error("Invalid TREF:", err.message);
      }
    }
  );
</script>

<style>
  /* Paste this once in your CSS */
  .tref-receiver { border: 2px dashed #5CCCCC; border-radius: 8px; padding: 20px; min-height: 80px; display: flex; align-items: center; justify-content: center; color: #6b7280; background: #f9fafb; transition: all 0.2s; }
  .tref-receiver-active { border-color: #8B5CF6; background: #f3e8ff; color: #8B5CF6; }
  .tref-receiver-success { border-color: #10B981; background: #ecfdf5; }
  .tref-receiver-has-block { border-style: solid; background: white; }
</style>

Bonus: Make Blocks Draggable

Add drag-and-drop to displayed blocks.

JavaScript
// After inserting the HTML
const element = document.querySelector('.tref-wrapper');
element.setAttribute('draggable', 'true');

element.addEventListener('dragstart', (e) => {
  wrapper.setDragData(e.dataTransfer);
  e.dataTransfer.effectAllowed = 'copy';
});

Complete Example

Full HTML page
<!DOCTYPE html>
<html>
<head>
  <title>My TREF Page</title>
  <script src="https://tref.lpmwfx.com/js/tref.min.js"></script>
  <style>
    /* Get all styles at once */
  </style>
</head>
<body>
  <h1>My Knowledge</h1>

  <!-- Block display -->
  <div id="my-block"></div>

  <!-- Drop zone -->
  <div id="dropzone">Drop TREF here</div>

  <script>
    (async () => {
      // Inject TREF styles
      const style = document.createElement('style');
      style.textContent = TREF.getStyles();
      document.head.appendChild(style);

      // Create and display a block
      const block = await TREF.publish("The mitochondria is the powerhouse of the cell.", {
        author: "Biology Teacher",
        refs: [{ type: "url", url: "https://en.wikipedia.org/wiki/Mitochondrion" }]
      });

      const wrapper = new TREF.TrefWrapper(block);
      const container = document.getElementById("my-block");
      container.innerHTML = wrapper.toHTML();

      // Make it draggable
      const el = container.querySelector('.tref-wrapper');
      el.setAttribute('draggable', 'true');
      el.addEventListener('dragstart', (e) => {
        wrapper.setDragData(e.dataTransfer);
      });

      // Set up receiver
      new TREF.TrefReceiver(document.getElementById("dropzone"), {
        onReceive: (w) => {
          document.getElementById("dropzone").innerHTML = w.toHTML();
        }
      });
    })();
  </script>
</body>
</html>
